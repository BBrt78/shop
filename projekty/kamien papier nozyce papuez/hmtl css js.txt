js

let userAnswer = document.querySelector(".user-answer");
let answerImg = document.querySelector(".answer-img");
let answer = document.querySelector(".answer")
let newGame = document.querySelector(".new-game");

let rock = document.querySelector(".rock-img");
rock.setAttribute("id", "0");
let paper = document.querySelector(".paper-img");
paper.setAttribute("id", "1");
let scissors = document.querySelector(".scissors-img");
scissors.setAttribute("id", "2");

let title = document.querySelector(".title")
let randomNumber = Math.floor(Math.random() * 3);

let rockImg = document.createElement("img");
rockImg.src = "img/rock.jpg";
rockImg.classList.add("hand");
rockImg.setAttribute("id", "0")
let paperImg = document.createElement("img");
paperImg.src = "img/paper.jpg";
paperImg.classList.add("hand");
paperImg.setAttribute("id", "1")
let scissorsImg = document.createElement("img");
scissorsImg.src = "img/scissors.jpg";
scissorsImg.classList.add("hand")
scissorsImg.setAttribute("id", "2")


let hands = [rockImg, paperImg, scissorsImg];


newGame.addEventListener("click", () => {
    
    userAnswer.classList.remove("hide");
});

[rock, paper, scissors].forEach(function(element) {
    element.addEventListener("click", e => {
        let randomNumber = Math.floor(Math.random() * 3);   //randomNumber - id odpowiedzi papueza
        title.innerText = "Odpowiedź Papueza:";
        answerImg.classList.add("hide");
        answer.appendChild(hands[randomNumber]);    //odpowiedz papueza
        //console.log(parseInt(element.id) + randomNumber);  //id tego co kliknal user

        //kamien papier nozyce
            if (parseInt(element.id) === 0 && randomNumber === 0) {
                title.innerText = "nie wiem"
            } else if(parseInt(element.id) === 0 && randomNumber === 1) {
                title.innerText = "jest możliwe"
            } else if(parseInt(element.id) === 0 && randomNumber === 2) {
                title.innerText = "co mam robić?"
            } else if(parseInt(element.id) === 1 && randomNumber === 0) {
                title.innerText = "co mam robić?"
            } else if(parseInt(element.id) === 1 && randomNumber === 1) {
                title.innerText = "nie wiem"
            } else if(parseInt(element.id) === 1 && randomNumber === 2) {
                title.innerText = "jest możliwe"
            } else if(parseInt(element.id) === 2 && randomNumber === 0) {
                title.innerText = "jest możliwe"
            } else if(parseInt(element.id) === 2 && randomNumber === 1) {
                title.innerText = "co mam robić?"
            } else if(parseInt(element.id) === 2 && randomNumber === 2) {
                title.innerText = "nie wiem"
            }
        

        newGame.addEventListener("click", () => {
            answer.removeChild(hands[randomNumber])
            
        });
        
    })
});

userAnswer.addEventListener("mouseover", () => {
    answerImg.classList.toggle("animation-papuez")
});
