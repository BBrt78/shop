js

let btn = document.querySelector(".input-btn");
let clear = document.querySelector(".clear-items");
let listItem = document.createElement("div");
let list = document.querySelector(".list");
let text = document.createElement("div");
let img1 = document.createElement("img");
let img2 = document.createElement("img");
let submitList = document.getElementById("submit-list");
let saveChanges = document.querySelector(".save-changes");
let id = 0;

btn.addEventListener("click", () => {
    listItem = document.createElement("div");
    list = document.querySelector(".list");
    text = document.createElement("div");
    img1 = document.createElement("img");
    img2 = document.createElement("img");
    img1.src = "img/edit.png";
    img2.src = "img/delete.png";
    img1.classList.add("edit");
    img2.classList.add("delete");

    listItem.appendChild(text);
    listItem.appendChild(img1);
    listItem.appendChild(img2);
    listItem.classList.add("list-item");
    list.appendChild(listItem);
    text.classList.add("item-text");
    text.textContent = document.getElementById("submit-list").value;
    localStorage.setItem("name", listItem.innerText);
    submitList.value = "";
    
    img1.value = text.innerText;
    img1.id = id;
    
    localStorage.setItem("id", id);
    let editImg = document.getElementById(id);
    console.log(localStorage)

    editImg.onclick = function editFunction() {
        submitList.value = editImg.value;
        
        editImg.addEventListener("mouseover", function btn() {
            editImg.value = submitList.value;

        })
        saveChanges.addEventListener("click", function onclick() {
            let y = submitList.value;
            text.innerText = y;
            submitList.value = "";
            });
    };

    img2.onclick = function deleteFunction() {

    };
    id++;
})

clear.addEventListener("click", () => {
    list.replaceChildren();
});
